--函数，输入人员姓名 返回人员年薪
create function hs1(spname varchar2) return
number is yearsal number(7,2);
begin
  --执行部分
  select sal*12+nvl(comm,0)*12 into yearsal from emp where ename=spname
  return yearsal;
  end;
